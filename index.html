-- LevelHUD.lua (LocalScript in StarterPlayerScripts) local Players = game:GetService("Players") local player = Players.LocalPlayer local playerGui = player:WaitForChild("PlayerGui") -- Build the screen GUI local screenGui = Instance.new("ScreenGui") screenGui.ResetOnSpawn = false screenGui.Parent = playerGui -- Coin counter local coinLabel = Instance.new("TextLabel") coinLabel.Size = UDim2.new(0, 200, 0, 50) coinLabel.Position = UDim2.new(0, 20, 0, 20) coinLabel.BackgroundColor3 = Color3.fromRGB(0, 0, 0) coinLabel.BackgroundTransparency = 0.5 coinLabel.TextColor3 = Color3.fromRGB(255, 210, 0) coinLabel.TextScaled = true coinLabel.Font = Enum.Font.GothamBold coinLabel.Text = "💰 Coins: 0" coinLabel.Parent = screenGui -- Win/notification overlay local notif = Instance.new("TextLabel") notif.Size = UDim2.new(0.6, 0, 0.2, 0) notif.Position = UDim2.new(0.2, 0, 0.4, 0) notif.BackgroundColor3 = Color3.fromRGB(0, 0, 0) notif.BackgroundTransparency = 0.3 notif.TextColor3 = Color3.fromRGB(0, 255, 120) notif.TextScaled = true notif.Font = Enum.Font.GothamBold notif.Text = "" notif.Visible = false notif.Parent = screenGui -- Listen for server events local evt = game:GetService("ReplicatedStorage"):WaitForChild("LevelEvents"):WaitForChild("LevelEvents") evt.OnClientEvent:Connect(function(eventType, value) if eventType == "COINS" then coinLabel.Text = "💰 Coins: " .. value elseif eventType == "WIN" then notif.Text = "🎉 LEVEL COMPLETE!\nCoins collected: " .. value notif.Visible = true end end)
